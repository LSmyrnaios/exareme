\begin{algorithm}
	\caption{\underline{\textsc{Naive Bayes Predict}}}
	\label{naivebayes_pred}
	\begin{algorithmic}[1] % The number tells where the line numbering should start
			\Procedure{Local}{$\mathbf{q}$, likelihood terms} \Comment{run for $l = 1, \dots, L$}
					\State $\hat{y} \gets \argmax_k \prod_l p(q_l | C_k)$
					\State \textbf{return} $\hat{y}$
			\EndProcedure
	\end{algorithmic}
\end{algorithm}
